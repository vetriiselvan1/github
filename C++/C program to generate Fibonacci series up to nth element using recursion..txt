#include <stdio.h> 
int fib(int n) { 
if (n == 0) return 0; 
if (n == 1) return 1; 
return fib(n - 1) + fib(n - 2); 
} 
int main() { 
int n, i; 
printf("Enter n (number of Fibonacci terms): "); 
scanf("%d", &n); 
if (n <= 0) { printf("Invalid n.\n"); return 0; } 
printf("Fibonacci series up to %d terms:\n", n); 
for (i = 0; i < n; i++) { 
printf("%d ", fib(i)); 
} 
printf("\n"); 
return 0; 
}
