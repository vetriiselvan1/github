// server.js
const express = require('express');
const http = require('http');
const socketIo = require('socket.io');
const cors = require('cors');

const app = express();
const server = http.createServer(app);
const io = socketIo(server, {
  cors: {
    origin: '*', // Allow all origins for demo purposes
    methods: ['GET', 'POST'],
  }
});

app.use(cors());
app.use(express.json());

// Sample in-memory messages
let messages = [
  { user: "Alice", text: "Hi everyone! Ready for todayâ€™s meeting?" },
  { user: "Me", text: "Yes, just joined the call." },
  { user: "Bob", text: "Can we discuss UI updates first?" },
  { user: "Me", text: "Absolutely!" }
];

// Basic API endpoint for messages
app.get('/api/messages', (req, res) => {
  res.json(messages);
});

// Real-time chat using Socket.IO
io.on('connection', (socket) => {
  console.log('New user connected');

  // Send all previous messages to newly connected client
  socket.emit('messages', messages);

  // Listen for sent messages
  socket.on('send_message', (msg) => {
    messages.push(msg);
    io.emit('new_message', msg); // Broadcast to all clients
  });

  socket.on('disconnect', () => {
    console.log('User disconnected');
  });
});

// Start server
const PORT = process.env.PORT || 4000;
server.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
